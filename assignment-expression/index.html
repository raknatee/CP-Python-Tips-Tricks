Reading a File with Generators 

from typing import Generator

def read_data_line_by_line()->Generator[str,None,None]:
    with open("./very-long-file.txt", "r") as data_file:
        while line:=data_file.readline():
            yield line

def main():
    for line in read_data_line_by_line():
        print(line.split(","))

main()
