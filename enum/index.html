from enum import Enum
class TrainingMode(Enum):
    BATCH = "BATCH"
    STOCHASTIC = "STOCHASTIC" 
    MINI_BATCH = "MINI_BATCH"
    
def train(training_mode:TrainingMode):
    if(training_mode == TrainingMode.BATCH):
        """
            do something       
        """
    if(training_mode == TrainingMode.STOCHASTIC):
        """
            do something
        """
    if(training_mode == TrainingMode.MINI_BATCH):
        """
            do something
        """
    
train(TrainingMode.BATCH)
train(TrainingMode.STOCHASTIC)
train(TrainingMode.MINI_BATCH)
